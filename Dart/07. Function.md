```dart

// Function declare
addTwoNumber(){
var x = 1;
var y = 2;
print(x+y);
}

// function call
addTwoNumber();



// Function receiving value
// Type casting or define like Sring x
addTwoNumber(var x, var y){
print(x+y);
}
//send parameter
addTwoNumber(4, 6);

// Return type int
Int addTwoNumber(var x, var y){
return x+y;
}
addTwoNumber(4, 6);


// Return noting
void addTwoNumber(var x, var y){
print(x+y);
}
addTwoNumber(4, 6);
```
